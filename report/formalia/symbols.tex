\newglossaryentry{TS}{
name=$T_S$,
description={day-ahead bid of energy production within a particular hour, made by the producer},
sort=t, type=symbols, symbol=[MW]
}

\newglossaryentry{PS}{
name=$P_S$,
description={energy spot price within a particular hour, as announced by the TSO the day ahead},
sort=p, type=symbols, symbol=[\euro/MWh]
}

\newglossaryentry{Pup}{
name=$P_{UP}$,
description={up-regulation price within a particular hour, determined by bid activations in the regulating market},
sort=p, type=symbols, symbol=[\euro/MWh]
}

\newglossaryentry{Pdn}{
name=$P_{DN}$,
description={down-regulation price within a particular hour, determined by bid activations in the regulating market},
sort=p, type=symbols, symbol=[\euro/MWh]
}

\newglossaryentry{Pess}{
name=$P_{ESS}$,
description={revenue made per energy unit stored in the ESS},
sort=p, type=symbols, symbol=[\euro/MWh]
}

\newglossaryentry{QG}{
name=$Q_G$,
description={power production sent to the grid within a particular hour},
sort=q, type=symbols, symbol=[MW]
}

\newglossaryentry{delta}{
name=$\Delta$,
description={deviation between bid and actual power production within a particular hour},
sort=d, type=symbols, symbol=[MW]
}

\newglossaryentry{delta+}{
name=$\Delta^+$,
description={positive part of deviation between bid and actual power production within a particular hour},
sort=d, type=symbols, symbol=[MW]
}

\newglossaryentry{delta-}{
name=$\Delta^-$,
description={negative part of deviation between bid and actual power production within a particular hour},

}

\newglossaryentry{Cdn}{
name=$C_{DN}$,
description={down-regulation penalty within a particular hour, defined as $P_S-P_{DN}$},
sort=c, type=symbols, symbol=[\euro/MWh]
}

\newglossaryentry{Cup}{
name=$C_{UP}$,
description={up-regulation penalty within a particular hour, defined as $P_{UP}-P_S$},
sort=c, type=symbols, symbol=[\euro/MWh]
}

\newglossaryentry{Cmin}{
name=$C_{min}$,
description={relative cost of storing energy in the ESS which is the minimum revenue that must be made on each unit of power for the ESS to be feasible},
sort=c, type=symbols, symbol=[\euro/MWh]
}
\newglossaryentry{N}{
