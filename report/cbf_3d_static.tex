\begin{equation}
\small
\dot{\begin{bmatrix}
	x\\y\\z
	\end{bmatrix}} =
\begin{bmatrix}
-1/\tau & 0 & 0\\0 & -1/\tau & 0 \\ 0 & 0 & -1/\tau
\end{bmatrix}
\begin{bmatrix}
x\\y\\z
\end{bmatrix} +
\begin{bmatrix}
1/\tau& 0 & 0 \\ 0& 1/\tau & 0 \\0& 0& 1/\tau
\end{bmatrix} 
\left(
\begin{bmatrix}
\bar{N} & 0 & 0 \\0 & \bar{N} & 0 \\0& 0& \bar{N}
\end{bmatrix}
\begin{bmatrix}
x_{ref}\\y_{ref}\\z_{ref}
\end{bmatrix}
-
\begin{bmatrix}
k_x & 0 & 0 \\0 & k_y & 0 \\0& 0&  k_z
\end{bmatrix}
\begin{bmatrix}
x\\y\\z
\end{bmatrix}
\right)
\end{equation}
